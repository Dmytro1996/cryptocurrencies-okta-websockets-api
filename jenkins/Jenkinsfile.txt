pipeline {
    action any
    stages {
        stage ("Test Pipeline"){
            steps {
                echo "Hello, world!"
            }
        }
        stage ("Test"){
            steps {
                mvn clean install
            }
        }
        stage ("Build"){
            steps {
                mvn clean install -DskipTests
            }
        }
        stage ("Deploy"){
            steps {
                mvn clean deploy -DmuleDeploy
            }
        }
    }
}
